import {Playground, PropsTable } from 'docz';
import Carousel from 'react-elastic-carousel';
import Item from '../components/SimpleItem';

- You can trigger the `goTo` method via a `ref` to the `Carousel`:  
```
this.carousel.goTo(someIndex)
```

<Playground >
    <label>Go to </label>
    <input type="number" defaultValue={0} onChange={({target}) => this.carousel.goTo(Number(target.value))} />
    <hr/>
    <Carousel ref={ref => this.carousel = ref}>
